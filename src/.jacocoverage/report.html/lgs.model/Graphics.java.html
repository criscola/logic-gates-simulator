/**

 * @author Dias
 */
public abstract class Graphics {
    protected Point origin;
    protected Point end;

    public Graphics(Point origin, Point end) {
        this.origin = origin;
        this.end = end;
    }

    public Point getEnd() {
        return end;
    }

    public Point getOrigin() {
        return origin;
    }

    public void setEnd(Point end) {
        this.end = end;
    }

    public void setOrigin(Point origin) {
        this.origin = origin;
    }
    
}
